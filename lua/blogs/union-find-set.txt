<p>　　并查集（Union-find set）用来表示一些不相交集合（Disjoint set），并且可以高效地实现以下两种操作：</p>
<ul>
	<li>查询：查询某个元素在哪个集合里</li>
	<li>合并：合并两个集合</li>
</ul>
<p>　　实现一般需要一个 father 数组，将集合实现成树，树的根就是这个集合的代表元。合并两个集合的时候，将其中一个的根节点的 father 设为另外一个。</p>
<p>　　并查集的关键就在于两个优化：</p>
<ol>
	<li>路径压缩（path compression）：因为我们只关心一个元素属于哪个集合，查找的时候可以把路径上的所有元素的父亲直接设为根，这样下次查找的时候就一步到位了。
<pre class="brush: cpp">
int find(int x) {
	if (father[x] != x) {
		father[x] = find(father[x]);
	}
	return father[x];
}
</pre></li>
	<li>按秩合并（union by rank）：或“启发式合并”。为了避免合并的时候出现某些畸形的情况，比如树退化成一个单链表，导致查询时间变成 O(n) 。我们可以再开一个数组 rank ，rank[i] （当 i 为根时才有意义）表示这棵树的高度。然后我们可以把高度小的树合并到高度大的树上，但又由于路径压缩，高度会改变，所以下面的 rank 相当于是曾经到过的最大高度：
<pre class="brush: cpp">
void union2(int x, int y) {
	int xRoot = find(x);
	int yRoot = find(y);
	if (xRoot == yRoot) {
		return;
	}
	if (rank[xRoot] &lt; rank[yRoot]) {
		father[xRoot] = yRoot;
	} else {
		father[yRoot] = xRoot;
		if (rank[xRoot] == rank[yRoot]) {
			rank[xRoot]++;
		}
	}
}
</pre>
若不等，小合并进大的，若相等，随意，且高度 + 1 。
</li>
</ol>
<p>　　以前我不理解并查集。集合嘛，用 STL 的 set 或者 Java 的 HashSet 不就行了？这些虽然也可以实现集合的并交补等运算，但并查集是专门用来高效实现“并”和“查”这两种运算的，单个操作的均摊时间复杂度为 O(α(n)) ，α 是 Ackermann 函数的反函数，α(n) 可以近似看成小于 5 的，所以相当于是 O(1) 。</p>

<dd>此题稍难，加上一个数组 delta 表示 i 与其父亲的关系：

<pre class="brush: cpp; collapse: true">
#include &lt;cstdio&gt;

using namespace std;

#define N 50001

int father[N];
int delta[N];
int rank[N];

/**
 * a, b are both in {-1, 0, 1}
 * 1 + 1 = -1
 * -1 + -1 = 1
 */
inline int add3(int a, int b) {
	return (a == b) ? (-a) : (a + b);
}

int find(int x) {
	if (father[x] != x) {
		int newfather = find(father[x]);
		delta[x] = add3(delta[x], delta[father[x]]);
		father[x] = newfather;
	}
	return father[x];
}

bool union2(int x, int y, int d) {
	int xr = find(x);
	int yr = find(y);
	if (xr == yr) {
		if (delta[x] != add3(d, delta[y])) {
			return false;
		}
		return true;
	}
	int tmp = add3(d, delta[y]);
	if (rank[xr] &lt; rank[yr]) {
		father[xr] = yr;
		delta[xr] = add3(tmp, -delta[x]);
	} else {
		father[yr] = xr;
		delta[yr] = add3(delta[x], -tmp);
		if (rank[xr] == rank[yr]) {
			rank[xr]++;
		}
	}
	return true;
}

int main(int argc, const char *argv[])
{
	int n, k;

		father[i] = i;

		delta[i] = 0;

		} else if (!union2(x, y, d - 1)) {

		}

<dt>poj 2236 Wireless Network</dt>
<dd>一个比较基础的并查集，但也有一些注意点：
<ol>
	<li>判断两点间距离小于某值，可以用平方，从而避免 double 和 sqrt ，浮点数比较大小是个坑</li>
	<li>我通过将 father[i] 设为 -1 来表示机器没有修好，一开始好几次 WA 。后来自己生成数据测试，马上发现了问题，查询的时候，必须判断 father[i] != -1 ，assert 帮了我很大的忙</li>

<pre class="brush: cpp; collapse: true">
#include &lt;cstdio&gt;
#include &lt;cassert&gt;

using namespace std;

const int N = 1010;

struct point_t {
	int x, y;
} points[N];

int father[N];
int rank[N];

int find(int x) {
	if (father[x] != x) {
		father[x] = find(father[x]);
	}
	return father[x];
}

void union2(int x, int y) {
	// printf("union %d, %d\n", x, y);
	int xr = find(x);
	int yr = find(y);
	if (xr == yr) {
		return;
	}
	if (rank[xr] &lt; rank[yr]) {
		father[xr] = yr;
	} else {
		father[yr] = xr;
		if (rank[xr] == rank[yr]) {
			rank[xr]++;
		}
	}
}

int distance2(int a, int b) {
	int dx = points[a].x - points[b].x;
	int dy = points[a].y - points[b].y;
	return dx * dx + dy * dy;
}

int main(int argc, const char *argv[])
{
	int n, d;
	scanf("%d%d", &amp;n, &amp;d);
	int d2 = d * d;
	for (int i = 1; i &lt;= n; i++) {
		scanf("%d%d", &amp;points[i].x, &amp;points[i].y);
		father[i] = -1;
		rank[i] = 0;
	}
	char op[4];
	while (scanf("%s", op) == 1) {
		if (op[0] == 'O') {
			int p;
			scanf("%d", &amp;p);
			if (father[p] == -1) {
				father[p] = p;
				// printf("%d repaired\n", p);
				for (int i = 1; i &lt;= n; i++) {
					if ((i != p) &amp;&amp; (father[i] != -1)) { // repaired
						if (distance2(i, p) &lt;= d2) { // in the area
							union2(i, p);
						}
					}
				}
			}
		} else if (op[0] == 'S') {
			int p, q;
			scanf("%d%d", &amp;p, &amp;q);
			// printf("testing %d and %d\n", p, q);
			if (father[p] != -1 &amp;&amp; father[q] != -1 &amp;&amp; find(p) == find(q)) {
				puts("SUCCESS");
			} else {
				puts("FAIL");
			}
		}
	}
	return 0;
}
